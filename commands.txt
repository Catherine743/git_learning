>git --version [command for checking git version]
>git config --list [git configuration list]
>git config --global user.name "username" [command for adding user name]
>git config --global user.email "useremail@gmail.com" [command for adding user email]
>git init [initializing git in present working directory]
>git status [list all untracked files]
>git add -A [command for adding all untracked files to staging area]
>git commit -m "commit message" [command for adding staged files to local repository]
>git log [command for listing commit history]
